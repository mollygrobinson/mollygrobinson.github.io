The month of March brings many great things: some warmer weather, leprechauns, and a ton of basketball.

Plus, it is arguably one of the best times of the year with the lovely March Madness tournament.

For this project, we were prompted with making a model to help us predict the outcome of games during the madness. I quickly learned no matter what metrics you pick, it’s literally impossible to get them all right, unless you’re just CRAZY lucky. For example, no one, or at least the vast majority of the population, would never pick Farleigh Dickinson over Purdue, or Princeton over Arizona.

For my predictors, I took a few different paths. I used my Big Ten tournament bracket as a learning experience and quickly learned that often less is more. First, I took possessions from each team, which helped see who usually takes control of the game more. Next, I looked at effective field goal shooting for each team, which was the team score divided by the possessions, then multiplied by 100. Another game changer for me was pace, or the cumulative mean for both offense and defense. This made a way different impact on the outcome of my games from my Big Ten Tournament predictions.

Then along came my model games, which included team and opponent rolling defense, team and opponent rolling offense, cumulative defense and opponent rolling possessions. Then I fed it all through the lightgbm feature.

Here’s the code block where I put everything into place.

library(tidyverse)
library(tidymodels)
library(hoopR)
library(zoo)
library(bonsai)
library(gt)
set.seed(1234)


teamgames <- load_mbb_team_box(seasons = 2015:2023) 

teamstats <- teamgames %>% 
  mutate(
    possessions = round(field_goals_attempted - offensive_rebounds + turnovers + (.475 * free_throws_attempted), digits = 0),
    team_score = free_throws_made + ((field_goals_made - three_point_field_goals_made)*2) + (three_point_field_goals_made*3),
    off = (team_score/possessions)*100,
    pos = cummean(possessions),
    offmean = cummean(off)
  ) %>% filter(possessions <= 90 & possessions >= 53)


rollingteamstats <- teamstats %>% 
  arrange(game_date) %>%
  group_by(team_short_display_name, season) %>%
  mutate(
    team_rolling_off = rollmean(lag(off, n=1), k=5, align="right", fill=NA),
    team_rolling_pos = rollmean(lag(pos, n=1), k=5, align="right", fill=NA),
    team_rolling_offmean = rollmean(lag(offmean, n=1), k=5, align="right", fill=NA)
    ) %>% 
  ungroup()


team_side <- rollingteamstats %>%
  select(
    game_id,
    team_id, 
    team_short_display_name, 
    opponent_team_id, 
    game_date, 
    season, 
    team_score,
    possessions,
    team_rolling_off,
    offensive_rebounds, 
    defensive_rebounds,
    team_rolling_pos,
    team_rolling_offmean
    )


opponent_side <- team_side %>%
  select(-opponent_team_id) %>% 
  rename(
    opponent_team_id = team_id,
    opponent_short_display_name = team_short_display_name,
    opponent_score = team_score,
    opponent_possessions = possessions,
    opponent_rolling_off = team_rolling_off,
    opponent_offensive_rebounds = offensive_rebounds, 
    opponent_defensive_rebounds = defensive_rebounds,
    opponent_rolling_pos = team_rolling_pos,
    opponent_rolling_offmean = team_rolling_offmean
  ) %>%
  mutate(opponent_team_id = as.numeric(opponent_team_id)
)


games <- team_side %>% inner_join(opponent_side)


teamdefensiveefficiency <- games %>% 
  arrange(game_date) %>% 
  group_by(team_short_display_name, season) %>% 
  mutate(
    team_def = (opponent_score/opponent_possessions)*100,
    team_rolling_def = rollmean(lag(team_def, n=1), k=15, align="right", fill=NA),
    team_cumulative_def = lag(cummean(team_def), n=1, default=0)
    ) %>%
  ungroup() %>%
  select(game_id, team_short_display_name, team_rolling_def, team_cumulative_def)



opponentdefensiveefficiency <- games %>% 
  arrange(game_date) %>% 
  group_by(opponent_short_display_name, season) %>% 
  mutate(
    opponent_def = (team_score/possessions)*100,
    opponent_rolling_def = rollmean(lag(opponent_def, n=1), k=15, align="right", fill=NA),
    opponent_cumulative_def = lag(cummean(opponent_def), n=1, default=0)
    ) %>%
  ungroup() %>%
  select(game_id, opponent_short_display_name, opponent_rolling_def, opponent_cumulative_def)
games <- games %>% inner_join(teamdefensiveefficiency)
games <- games %>% inner_join(opponentdefensiveefficiency)
games <- games %>% mutate(
  team_result = as.factor(case_when(
    team_score > opponent_score ~ "W",
    opponent_score > team_score ~ "L"
))) %>% na.omit()

games$team_result <- relevel(games$team_result, ref="W")


modelgames <- games %>%
  select(
    game_id,
    game_date,
    team_short_display_name,
    opponent_short_display_name,
    season,
    team_result,
    team_rolling_def,
    opponent_rolling_def,
    team_rolling_offmean,
    opponent_rolling_offmean,
    team_cumulative_def,
    opponent_cumulative_def,
    team_rolling_pos,
    opponent_rolling_pos
    ) %>% na.omit()


game_split <- initial_split(modelgames, prop = .8)
log_train <- training(game_split)
log_test <- testing(game_split)


lightgbm_mod <- 
  boost_tree() %>%
  set_engine("lightgbm") %>%
  set_mode(mode = "classification")


game_recipe <- 
  recipe(team_result ~ ., data = log_train) %>% 
  update_role(game_id, game_date, team_short_display_name, opponent_short_display_name, season, new_role = "ID") %>%
  step_normalize(all_predictors())


lightgbm_workflow <- 
  workflow() %>% 
  add_model(lightgbm_mod) %>% 
  add_recipe(game_recipe)


lightgbm_fit <- 
  lightgbm_workflow %>% 
  fit(data = log_train)
